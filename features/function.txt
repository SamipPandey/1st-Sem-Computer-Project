        if (board[i][0] == board[i][1] &&
            board[i][1] == board[i][2])

            board[1][i] == board[2][i])
            return 1;

    if (board[0][0] == board[1][1] &&
        board[1][1] == board[2][2])
        return 1;

    if (board[0][2] == board[1][1] &&
        board[1][1] == board[2][0])
